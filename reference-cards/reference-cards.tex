\sect{Line Editing}
\begin{keys}
\texttt{C-a / C-e}    & Line start / end \\
\texttt{M-b / M-f}    & Word back / forward \\
\texttt{C-w}          & Delete word backward \\
\texttt{M-d}          & Delete word forward \\
\texttt{C-u}          & Delete to line start \\
\texttt{C-k}          & Delete to line end \\
\texttt{C-y}          & Paste deleted text \\
\texttt{C-r}          & History search (fzf) \\
\texttt{C-t}          & File picker (fzf) \\
\texttt{M-c}          & Fuzzy cd (fzf + fd) \\
\texttt{Tab}          & Complete (menu) \\
\texttt{\textrightarrow}  & Accept suggestion \\
\texttt{M-\textrightarrow} & Accept next word \\
\end{keys}

\sect{History \normalfont(8192, shared sessions)}
\begin{keys}
\texttt{!!}                & Repeat last command \\
\texttt{!\$}               & Last argument \\
\texttt{!abc}              & Last cmd starting \texttt{abc} \\
\texttt{!abc:p}            & Print don't execute \\
\texttt{history | rg pat}  & Search history \\
\end{keys}

\columnbreak

\sect{Job Control}
\begin{keys}
\texttt{C-z}       & Suspend foreground \\
\texttt{bg}        & Resume in background \\
\texttt{fg}        & Resume in foreground \\
\texttt{jobs}      & List background jobs \\
\texttt{kill \%1}  & Kill job 1 \\
\end{keys}

\sect{Redirection \& Piping}
\begin{keys}
\texttt{cmd \&}          & Run in background \\
\texttt{c1 \&\& c2}     & Run c2 if c1 succeeds \\
\texttt{c1 || c2}       & Run c2 if c1 fails \\
\texttt{c1 | c2}        & Pipe stdout \\
\texttt{c1 |\& c2}      & Pipe stdout+stderr \\
\texttt{cmd > file}      & Redirect stdout \\
\texttt{cmd 2>\&1}       & Stderr to stdout \\
\texttt{cmd \&> file}    & Redirect all \\
\texttt{cmd >> file}     & Append \\
\end{keys}

Plugins: autosuggestions, syntax-highlighting}

\sect{Zsh Globbing}
\begin{keys}
\texttt{**/*.py}       & Recursive glob \\

\texttt{*(/)}          & Directories only \\

\texttt{*(Lk+100)}     & Files > 100KB \\
\texttt{*(@)}          & Symlinks only \\
\end{keys}

\sect{Parameter Expansion}
\begin{keys}
\texttt{\$\{v:-default\}} & Default if unset \\
\texttt{\$\{v:=default\}} & Assign if unset \\
\texttt{\$\{v\#pat\}}     & Remove shortest prefix \\
\texttt{\$\{v\#\#pat\}}   & Remove longest prefix \\
\texttt{\$\{v\%pat\}}     & Remove shortest suffix \\
\texttt{\$\{v\%\%pat\}}   & Remove longest suffix \\
\texttt{\$\{v/old/new\}}  & Replace first \\
\texttt{\$\{v//old/new\}} & Replace all \\
\texttt{\$\{v:u\} / \$\{v:l\}} & Upper / lower \\
\texttt{\$\{\#v\}}        & Length \\
\end{keys}
