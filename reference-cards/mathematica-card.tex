\begin{multicols}{3}

\sect{Basic Syntax}
\begin{tabularx}{\linewidth}{@{}l@{\enspace}Y@{}}
\texttt{f[x, y]}                    & Function call \\
\texttt{x = expr}                   & Assignment \\
\texttt{x := expr}                  & Delayed assignment \\
\texttt{f[x\_] := body}            & Function definition \\
\texttt{x /. rule}                  & Apply rule \\
\texttt{x //. rule}                 & Apply repeatedly \\
\texttt{expr // f}                  & Postfix apply \\
\texttt{f @ x}                      & Prefix apply \\
\texttt{f @@ list}                  & Apply to list \\
\texttt{f @@@ list}                 & Apply at level 1 \\
\texttt{x // f // g}               & Pipeline \\
\texttt{(* comment *)}             & Comment \\
\end{tabularx}

\sect{Patterns \& Rules}
\begin{tabularx}{\linewidth}{@{}l@{\enspace}Y@{}}
\texttt{\_}                          & Any expression \\
\texttt{x\_}                        & Named pattern \\
\texttt{\_Integer}                  & Head pattern \\
\texttt{x\_?NumericQ}              & Conditional pattern \\
\texttt{\_\_} / \texttt{\_\_\_}     & One+ / zero+ args \\
\texttt{x\_ /; cond}               & Pattern with guard \\
\texttt{a -> b}                     & Rule \\
\texttt{a :> b}                     & Delayed rule \\
\texttt{Cases[xs, pat]}            & Extract matches \\
\texttt{MatchQ[expr, pat]}         & Test pattern \\
\texttt{Replace[e, rules]}         & Apply rules \\
\end{tabularx}

\sect{Lists}
\begin{tabularx}{\linewidth}{@{}l@{\enspace}Y@{}}
\texttt{\{a, b, c\}}               & List \\
\texttt{xs[[i]]}                    & Element (1-based) \\
\texttt{xs[[i;;j]]}                & Span / slice \\
\texttt{xs[[-1]]}                   & Last element \\
\texttt{Length[xs]}                 & Length \\
\texttt{Join[a, b]}                & Concatenate \\
\texttt{Append[xs, x]}             & Append \\
\texttt{Prepend[xs, x]}            & Prepend \\
\texttt{Flatten[xs]}               & Flatten nested \\
\texttt{Partition[xs, n]}          & Chunk by n \\
\texttt{Range[a, b, step]}         & Number range \\
\texttt{Table[f[i], \{i, n\}]}    & Generate list \\
\end{tabularx}

\sect{Functional}
\begin{tabularx}{\linewidth}{@{}l@{\enspace}Y@{}}
\texttt{Map[f, xs]}                 & Apply f to each \\
\texttt{f /@ xs}                    & Map (operator) \\
\texttt{Select[xs, p]}             & Filter \\
\texttt{Fold[f, init, xs]}         & Left fold \\
\texttt{FoldList[f, init, xs]}     & Scan / accumulate \\
\texttt{MapThread[f, \{a, b\}]}   & Zip + map \\
\texttt{Thread[f[a, b]]}           & Thread over lists \\
\texttt{Nest[f, x, n]}             & Apply f n times \\
\texttt{NestList[f, x, n]}         & Collect iterations \\
\texttt{FixedPoint[f, x]}          & Iterate to fixpoint \\
\texttt{Function[\{x\}, body]}     & Named pure fn \\
\texttt{\#1, \#2}                   & Slot (arg 1, 2) \\
\texttt{f[\#] \&}                   & Pure function \\
\texttt{Composition[f, g]}         & $f \circ g$ \\
\texttt{Identity}                   & Identity function \\
\end{tabularx}

\sect{Control Flow}
\begin{tabularx}{\linewidth}{@{}l@{\enspace}Y@{}}
\texttt{If[cond, then, else]}      & Conditional \\
\texttt{Which[c1,v1, c2,v2]}      & Multi-branch \\
\texttt{Switch[x, p1,v1, ...]}    & Pattern switch \\
\texttt{Do[body, \{i, n\}]}       & For loop \\
\texttt{While[cond, body]}         & While loop \\
\texttt{Module[\{x=v\}, body]}    & Local scope \\
\texttt{Block[\{x\}, body]}       & Dynamic scope \\
\texttt{With[\{x=v\}, body]}      & Constants \\
\end{tabularx}

\sect{Associations (Dicts)}
\begin{tabularx}{\linewidth}{@{}l@{\enspace}Y@{}}
\texttt{<|"a"->1, "b"->2|>}       & Association \\
\texttt{assoc["key"]}              & Lookup \\
\texttt{Keys[a] / Values[a]}      & Keys / values \\
\texttt{Merge[a, b]}              & Merge \\
\texttt{KeySelect[a, p]}          & Filter by key \\
\texttt{AssociationMap[f, a]}     & Map over assoc \\
\end{tabularx}

\sect{String Operations}
\begin{tabularx}{\linewidth}{@{}l@{\enspace}Y@{}}
\texttt{StringJoin[a, b]} / \texttt{a <> b}  & Concatenate \\
\texttt{StringSplit[s, ","]}       & Split \\

\texttt{StringMatchQ[s, pat]}      & Pattern test \\

\texttt{ToString[expr]}            & Convert to string \\
\texttt{StringTemplate["<*x*>"]}  & Template \\
\end{tabularx}

\sect{Debugging}

\texttt{Echo[expr]}                & Print and return \\
