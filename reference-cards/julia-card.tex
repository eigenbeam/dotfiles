Run                  & \texttt{julia script.jl}                           & \texttt{-{}-project=.} for local env \\
REPL                 & \texttt{julia}                                     & \texttt{]} pkg, \texttt{;} shell, \texttt{?} help modes \\
Test                 & \texttt{] test}                                    & runs \texttt{test/runtests.jl} \\
Format               & \texttt{JuliaFormatter.format\_file()}             & \texttt{] add JuliaFormatter} \\
Lint                 & ---                                                & LSP provides diagnostics \\
LSP                  & \texttt{LanguageServer.jl}                         & configured in neovim via lspconfig \\

% ── Language-Specific Content ──────────────────────────────────────
\begin{multicols}{3}

\sect{Basic Syntax}
\begin{tabularx}{\linewidth}{@{}l@{\enspace}Y@{}}
\texttt{function f(x) ... end}      & Function \\
\texttt{f(x) = expr}                & Short form \\
\texttt{x -> expr}                   & Anonymous function \\
\texttt{if / elseif / else / end}   & Conditional \\
\texttt{x ? a : b}                   & Ternary \\
\texttt{for x in xs ... end}        & For loop \\
\texttt{while cond ... end}         & While loop \\
\texttt{begin ... end}               & Block expression \\
\texttt{let x = v ... end}          & Local scope \\
\texttt{return expr}                 & Return value \\
\end{tabularx}

\sect{Types \& Dispatch}
\begin{tabularx}{\linewidth}{@{}l@{\enspace}Y@{}}
\texttt{struct S ... end}            & Immutable struct \\
\texttt{mutable struct S}            & Mutable struct \\
\texttt{abstract type T end}         & Abstract type \\
\texttt{T <: S}                      & Subtype \\
\texttt{f(x::Int)}                   & Type annotation \\
\texttt{f(x::T) where T}            & Parametric method \\
\texttt{methods(f)}                  & List methods \\
\texttt{@which f(x)}                & Which method? \\
\texttt{isa(x, T)}                   & Type check \\
\texttt{typeof(x)}                   & Runtime type \\
\texttt{Union\{A, B\}}              & Union type \\
\texttt{Nothing}                     & Unit / void type \\
\end{tabularx}

\sect{Common Types}
\begin{tabularx}{\linewidth}{@{}l@{\enspace}Y@{}}
\texttt{Int64, Float64}              & Numeric defaults \\
\texttt{Bool}                        & true / false \\
\texttt{String, Char}               & Text \\
\texttt{Symbol} (\texttt{:name})    & Interned name \\
\texttt{Nothing}                     & Singleton \\
\texttt{Missing}                     & Missing data \\
\texttt{Tuple\{A,B\}}               & Fixed tuple \\
\texttt{NamedTuple}                  & Named tuple \\
\texttt{Dict\{K,V\}}                & Dictionary \\
\texttt{Set\{T\}}                   & Set \\
\end{tabularx}

\sect{Arrays}
\begin{tabularx}{\linewidth}{@{}l@{\enspace}Y@{}}
\texttt{[1, 2, 3]}                   & Vector (1-indexed) \\
\texttt{[1 2; 3 4]}                  & Matrix \\
\texttt{zeros(m, n)}                 & Zero matrix \\
\texttt{ones(m, n)}                  & Ones matrix \\
\texttt{fill(v, m, n)}              & Fill value \\
\texttt{a[i]}                        & Index (1-based) \\
\texttt{a[2:5]}                      & Slice \\
\texttt{a[end]}                      & Last element \\
\texttt{push!(a, x)}                & Append (mutate) \\
\texttt{length(a)}                   & Length \\
\texttt{size(a)}                     & Dimensions \\
\texttt{a'}                          & Adjoint (transpose) \\
\texttt{vcat(a, b) / hcat(a, b)}   & Stack vert / horiz \\
\texttt{vec(a)}                      & Flatten to vector \\
\end{tabularx}

\sect{Comprehensions \& Generators}
\begin{tabularx}{\linewidth}{@{}l@{\enspace}Y@{}}
\texttt{[f(x) for x in xs]}         & Array comprehension \\
\texttt{[f(x) for x in xs if p]}   & Filtered \\
\texttt{[f(i,j) for i=1:m, j=1:n]} & 2D comprehension \\
\texttt{Dict(k => v for ...)}       & Dict comprehension \\
\texttt{(f(x) for x in xs)}         & Generator (lazy) \\
\end{tabularx}

\sect{Strings \& I/O}
\begin{tabularx}{\linewidth}{@{}l@{\enspace}Y@{}}
\texttt{"\$x and \$(expr)"}         & String interpolation \\
\texttt{string(a, b)}               & Concatenate \\
\texttt{join(xs, ",")}              & Join with sep \\
\texttt{split(s, ",")}              & Split string \\
\texttt{strip(s)}                    & Trim whitespace \\
\texttt{println(x)}                 & Print with newline \\
\texttt{@show x}                    & Debug print \\
\texttt{@info "msg" x y}           & Logging \\
\texttt{open(f, "r") do io}        & File handle \\
\texttt{read(f, String)}           & Read file \\
\texttt{write(f, data)}            & Write file \\
\end{tabularx}

\sect{Functional Patterns}
\begin{tabularx}{\linewidth}{@{}l@{\enspace}Y@{}}
\texttt{map(f, xs)}                 & Apply f to each \\
\texttt{filter(p, xs)}             & Keep matching \\
\texttt{reduce(f, xs)}             & Fold \\
\texttt{mapreduce(f, op, xs)}      & Map then reduce \\
\texttt{any(p, xs) / all(p, xs)}   & Existential / universal \\
\texttt{zip(xs, ys)}               & Pair elements \\
\texttt{enumerate(xs)}             & Index + element \\
\texttt{sort(xs; by=f)}            & Sort by projection \\
\texttt{unique(xs)}                & Unique values \\
\texttt{xs .|> f}                  & Broadcast pipe \\
\texttt{x |> f |> g}              & Pipeline \\
\texttt{f} $\circ$ \texttt{g}      & Compose (unicode) \\
\texttt{do ... end}                & Closure block \\
\end{tabularx}

\sect{Broadcasting}
\begin{tabularx}{\linewidth}{@{}l@{\enspace}Y@{}}
\texttt{f.(xs)}                     & Broadcast call \\
\texttt{a .+ b}                    & Elementwise ops \\
\texttt{@. expr}                   & Broadcast everything \\
\texttt{a .== b}                   & Elementwise compare \\
\end{tabularx}

\sect{Error Handling}
\begin{tabularx}{\linewidth}{@{}l@{\enspace}Y@{}}
\texttt{try ... catch e ... end}   & Try/catch \\
\texttt{finally ... end}           & Always runs \\
\texttt{throw(ErrorException(m))}  & Throw \\
\texttt{error("msg")}             & Shorthand throw \\
\texttt{@assert cond "msg"}       & Assertion \\
\end{tabularx}

\sect{Macros \& Metaprogramming}
\begin{tabularx}{\linewidth}{@{}l@{\enspace}Y@{}}
\texttt{@time expr}                & Benchmark \\
\texttt{@benchmark expr}          & BenchmarkTools \\
\texttt{@code\_warntype f(x)}     & Type stability \\
\texttt{@macroexpand @m x}        & Expand macro \\
\texttt{macro m(x) ... end}       & Define macro \\
\texttt{:(expr)}                   & Quote expression \\
\texttt{\$(expr)}                  & Interpolate in quote \\
\end{tabularx}

\end{multicols}
