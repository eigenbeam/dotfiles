\cardtitle{Claude Code --- Configuration \& Hooks}

\begin{multicols}{3}

\sect{CLAUDE.md Files}
\begin{tabularx}{\linewidth}{@{}l@{\enspace}Y@{}}
\texttt{./CLAUDE.md}                 & Project (shared via git) \\
\texttt{./.claude/CLAUDE.md}        & Alt project location \\
\texttt{\~{}/.claude/CLAUDE.md}     & User (all projects) \\
\texttt{.claude/rules/*.md}         & Scoped rules \\
\texttt{@path/to/file}              & Import syntax \\
\end{tabularx}
{\footnotesize Rules support \texttt{paths:} frontmatter to scope to specific files. Keep CLAUDE.md $<$200 lines.}

\sect{Settings Files}
\begin{tabularx}{\linewidth}{@{}l@{\enspace}Y@{}}
\texttt{\~{}/.claude/settings.json}  & User settings \\
\texttt{.claude/settings.json}       & Project (shared) \\
\texttt{.claude/settings.local.json} & Project (local, gitignored) \\
\end{tabularx}
{\footnotesize Precedence: user $<$ project $<$ local $<$ CLI args $<$ managed.}

\sect{Settings Keys}
\begin{tabularx}{\linewidth}{@{}l@{\enspace}Y@{}}
\texttt{model}                       & Default model alias \\
\texttt{effortLevel}                 & low / medium / high \\
\texttt{autoMemoryEnabled}          & Auto-memory on/off \\
\texttt{permissions.defaultMode}    & Default perm mode \\
\texttt{permissions.allow}          & Auto-approve list \\
\texttt{permissions.deny}           & Block list \\
\texttt{hooks}                       & Hook definitions \\
\texttt{env}                         & Environment vars \\
\texttt{sandbox.enabled}            & Sandbox mode \\
\end{tabularx}

\sect{Permission Rules}
\begin{tabularx}{\linewidth}{@{}l@{\enspace}Y@{}}
\texttt{"Bash"}                      & Entire tool \\
\texttt{"Bash(npm run *)"}          & Glob on command \\
\texttt{"Bash(git push *)"}         & In deny list \\
\texttt{"Edit(src/**/*.ts)"}        & Glob on file path \\
\texttt{"Read(\~{}/.ssh/*)"}        & Block sensitive files \\
\texttt{"WebFetch(domain:gh.com)"}  & Domain filter \\
\texttt{"mcp\_\_server\_\_*"}       & MCP tool pattern \\
\end{tabularx}
{\footnotesize Eval order: deny $\to$ ask $\to$ allow.}

\sect{Permission Modes}
\begin{tabularx}{\linewidth}{@{}l@{\enspace}Y@{}}
\texttt{default}                     & Prompt on first use \\
\texttt{plan}                        & Read-only analysis \\
\texttt{acceptEdits}                 & Auto-accept edits \\
\texttt{dontAsk}                     & Deny unless allowed \\
\texttt{bypassPermissions}          & Skip all prompts \\
\end{tabularx}

\sect{Hook Events}
\begin{tabularx}{\linewidth}{@{}l@{\enspace}Y@{}}
\texttt{PreToolUse}                  & Before tool (can block) \\
\texttt{PostToolUse}                 & After tool success \\
\texttt{UserPromptSubmit}           & User sends prompt \\
\texttt{SessionStart}               & Session begins \\
\texttt{SessionEnd}                  & Session ends \\
\texttt{Stop}                        & Claude finishes \\
\texttt{Notification}               & Needs attention \\
\texttt{SubagentStart/Stop}         & Subagent lifecycle \\
\texttt{PreCompact/PostCompact}     & Context compaction \\
\end{tabularx}

\sect{Hook Types}
\begin{tabularx}{\linewidth}{@{}l@{\enspace}Y@{}}
\texttt{command}                     & Run shell command \\
\texttt{http}                        & POST to endpoint \\
\texttt{prompt}                      & Single-turn LLM \\
\texttt{agent}                       & Multi-turn subagent \\
\end{tabularx}
{\footnotesize Exit codes: 0 = allow, 2 = block.}

\sect{Hook Example}
{\footnotesize
\begin{verbatim}
"hooks": {
  "PostToolUse": [{
    "matcher": "Edit|Write",
    "hooks": [{
      "type": "command",
      "command": "prettier --write
        \"$TOOL_INPUT.file_path\""
    }]
  }]
}
\end{verbatim}
}

\sect{MCP Configuration}
\begin{tabularx}{\linewidth}{@{}l@{\enspace}Y@{}}
\texttt{\~{}/.claude/mcp.json}      & User-scope servers \\
\texttt{.claude/.mcp.json}           & Project-scope \\
\texttt{-{}-mcp-config f.json}      & Session-scope \\
\texttt{claude mcp add \textit{n}}  & Add server (CLI) \\
\texttt{claude mcp remove \textit{n}} & Remove server \\

\sect{Headless Patterns}
{\footnotesize
\begin{verbatim}
# JSON output
claude -p "query" \
  --output-format json | jq '.result'

# Auto-approve tools
claude -p "query" \
  --allowedTools "Bash,Read,Edit"

# Pipe input
cat f.txt | claude -p "analyze"

# Continue session
claude -p "follow-up" \
  --resume $SESSION_ID
\end{verbatim}
}
