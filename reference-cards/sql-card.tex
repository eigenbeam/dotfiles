% ── Standard Tasks ─────────────────────────────────────────────────
\begin{tabularx}{\linewidth}{@{}l@{\quad}l@{\quad}Y@{}}
\textbf{Task}        & \textbf{Command}                                  & \textbf{Notes} \\[2pt]
Install              & \texttt{brew install postgresql@16}                & \texttt{brew services start postgresql@16} \\
CLI client           & \texttt{psql -d \textit{dbname}}                  & \texttt{-U user -h host -p port} \\
Create database      & \texttt{createdb \textit{name}}                   & or \texttt{CREATE DATABASE name;} \\
Drop database        & \texttt{dropdb \textit{name}}                     & or \texttt{DROP DATABASE name;} \\
Dump                 & \texttt{pg\_dump \textit{db} > f.sql}            & \texttt{-Fc} for custom format \\
Restore              & \texttt{pg\_restore -d \textit{db} f.dump}       & or \texttt{psql -f f.sql} \\
GUI                  & \texttt{pgcli} / DBeaver                          & \texttt{brew install pgcli} \\
\end{tabularx}
