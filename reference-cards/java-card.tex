\sect{Basic Syntax}
\begin{tabularx}{\linewidth}{@{}l@{\enspace}Y@{}}
\texttt{var x = expr}                & Local type inference \\
\texttt{final var x = expr}          & Immutable local \\
\texttt{if / else if / else}         & Conditional \\
\texttt{switch (x) \{ case ... \}}   & Switch statement \\
\texttt{case P -> expr}              & Switch expression (14+) \\
\texttt{for (var x : xs)}            & Enhanced for \\
\texttt{while / do-while}            & Loops \\
\texttt{try / catch / finally}       & Error handling \\
\texttt{throw new E("msg")}          & Throw exception \\
\texttt{import static}               & Static import \\
\end{tabularx}

\sect{Types \& Data}
\begin{tabularx}{\linewidth}{@{}l@{\enspace}Y@{}}
\texttt{record R(int x, String y)}   & Record (16+) \\
\texttt{sealed interface I}          & Sealed type (17+) \\
\texttt{permits A, B}               & Permitted subtypes \\
\texttt{enum E \{ A, B, C \}}       & Enum \\
\texttt{interface I}                 & Interface \\
\texttt{class C extends B}           & Inheritance \\
\texttt{implements I}                & Implement interface \\
\texttt{Optional<T>}                & Nullable wrapper \\
\texttt{@Override}                   & Override annotation \\
\end{tabularx}

\sect{Modern Java (17+)}
\begin{tabularx}{\linewidth}{@{}l@{\enspace}Y@{}}
\texttt{case P when g ->}            & Guarded pattern (21+) \\
\texttt{instanceof P p}              & Pattern match (16+) \\
\texttt{""" text block """}          & Multiline string (15+) \\
\texttt{yield expr}                  & Switch yield \\
\texttt{Stream.toList()}             & Unmodifiable list (16+) \\
\texttt{SequencedCollection}         & Ordered access (21+) \\
\texttt{.getFirst() / .getLast()}    & First/last element (21+) \\
\end{tabularx}

\sect{Generics}
\begin{tabularx}{\linewidth}{@{}l@{\enspace}Y@{}}
\texttt{class C<T>}                  & Generic class \\
\texttt{<T> T f(T x)}               & Generic method \\
\texttt{<T extends Comparable>}     & Upper bound \\
\texttt{List<? extends T>}          & Covariant \\
\texttt{List<? super T>}            & Contravariant \\
\texttt{List<?>}                     & Wildcard \\
\end{tabularx}

\sect{Collections}
\begin{tabularx}{\linewidth}{@{}l@{\enspace}Y@{}}
\texttt{List.of(a, b, c)}           & Immutable list \\
\texttt{Set.of(a, b)}               & Immutable set \\
\texttt{Map.of(k1,v1, k2,v2)}      & Immutable map \\
\texttt{new ArrayList<>()}           & Mutable list \\
\texttt{new HashMap<>()}             & Mutable map \\
\texttt{new HashSet<>()}             & Mutable set \\
\texttt{Collections.unmodifiable*}  & Unmodifiable view \\
\texttt{Deque} / \texttt{ArrayDeque} & Stack / queue \\
\end{tabularx}

\texttt{.filter(p)}                  & Filter \\
\texttt{.flatMap(f)}                 & Flatten \\

\texttt{.collect(toList())}          & Collect to list \\

\texttt{.sorted(comparator)}        & Sort \\

\texttt{.limit(n) / .skip(n)}       & Take / drop \\
\texttt{.anyMatch(p)}               & Existential \\
\texttt{.allMatch(p)}               & Universal \\
\texttt{.findFirst()}               & First as Optional \\
\texttt{Stream.of(a, b)}            & From values \\

\cardtitle{Java --- Concurrency \& Standard Library}

\begin{multicols}{3}

\sect{Concurrency --- Basics}
\begin{tabularx}{\linewidth}{@{}l@{\enspace}Y@{}}

\texttt{synchronized}                  & Mutual exclusion \\
\texttt{volatile}                      & Visibility guarantee \\
\texttt{Thread.sleep(ms)}              & Sleep \\
\texttt{thread.join()}                 & Wait for completion \\
\end{tabularx}

\sect{Concurrency --- Executors}
\begin{tabularx}{\linewidth}{@{}l@{\enspace}Y@{}}

\texttt{executor.submit(task)}          & Submit task \\
\texttt{Future<T>}                     & Async result \\
\texttt{future.get()}                   & Block for result \\
\end{tabularx}

\sect{CompletableFuture}
\begin{tabularx}{\linewidth}{@{}l@{\enspace}Y@{}}
\texttt{supplyAsync(() -> v)}          & Async supplier \\
\texttt{.thenApply(f)}                  & Map result \\
\texttt{.thenCompose(f)}                & FlatMap \\
\texttt{.thenAccept(c)}                 & Consume result \\
\texttt{.exceptionally(f)}              & Handle error \\
\texttt{.thenCombine(other, f)}         & Combine two \\
\texttt{allOf(a, b, c)}                 & Wait for all \\
\texttt{anyOf(a, b, c)}                 & First completed \\
\texttt{.join()}                         & Block (unchecked) \\
